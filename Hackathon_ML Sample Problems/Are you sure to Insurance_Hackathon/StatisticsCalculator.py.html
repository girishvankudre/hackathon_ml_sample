199      def doLabelEncoding(self,df,eligible_for_encoding_catcols): 
200          df2 = df.copy() 
201          for x in eligible_for_encoding_catcols: 
202              df2[x] = df2[x].astype('category') 
203              newcol = x + "_encoded" 
204              df2[newcol] = df2[x].cat.codes 
205              df2.drop(x, axis=1, inplace=True) 
206   
207          #print("Returning Label encoded DataFrame with columns:\n", df2.dtypes) 
208          return df2

217      def drop_numerical_outliers(self, df2, z_thresh=3): 
218          # Constrains will contain `True` or `False` depending on if it is a value below the threshold. 
219          print("in drop_numerical_outliers") 
220          df = pd.DataFrame(df2) 
221          print("shape before ", df.shape) 
222          constrains = df.select_dtypes(include=[np.number]) \ 
223              .apply(lambda x: np.abs(stats.zscore(x)) < z_thresh, result_type='reduce') \ 
224              .all(axis=1) 
225          # Drop (inplace) values set to be rejected 
226          df.drop(df.index[~constrains], inplace=True) 
227          print("shape after ",df.shape) 
228          return df

413      def getFeaturesAndRankingsUsingRFEandLogisticRegression(self,basedf,targetcol,flag): 
414   
415          print("Getting Features using RFE + LOG REG with targetcol :", targetcol, " : flag :", flag) 
416   
417          df = basedf.copy() 
418          X = df.drop(targetcol, axis=1)  # Feature Matrix 
419          y = df[targetcol]  # dependent variable 
420          regressor = LogisticRegression(solver='lbfgs',max_iter=900) 
421          # rank all features, i.e continue the elimination until the last one 
422          rfe = RFE(regressor, n_features_to_select=5) 
423          rfe.fit(X, y) 
424          featurecols=list(X.columns) 
425          rfefeatures=rfe.n_features_ 
426          rfesupport=rfe.support_ 
427          rferanking=rfe.ranking_ 
428          rferesults = list(zip(featurecols, rferanking, rfesupport)) 
429          self.init_rfe_log_reg_df = pd.DataFrame(rferesults) 
430          self.init_rfe_log_reg_df.name = flag+" RFE LOG REG" 
431          print("RFE Results using Logistic Regression : \n", self.init_rfe_log_reg_df) 
432          return self.init_rfe_log_reg_df

449      def drawPie(self,column,chartname): 
450          ''' 
451          To be used ideally only for categorical data , i.e < 10 distinct values 
452          :param nparr: numpy array column 
453          :return: none 
454          ''' 
455          nparr = np.array(column) 
456          uniqueValues, occurCount = np.unique(nparr, return_counts=True) 
457          uniqueCount = len(uniqueValues) 
458          totalCount = np.sum(occurCount) 
459          print("Drawing a pie chart ") 
460          print("Unique Values : ", uniqueValues) 
461          print("Occurrence Count : ", occurCount) 
462          print("Total count :", totalCount) 
463          chartfilename=chartname+".pdf" 
464          # print ("# of unique values :",len(uniqueValues)) 
465   
466          if (uniqueCount >= 10): 
467              print(" Too many values : not suited for pie chart ") 
468          else: 
469              labels = uniqueValues 
470              sizes = occurCount 
471   
472              explode = (0, 0.1, 0, 0, 0)  # only "explode" the 2nd slice (i.e. 'Hogs') 
473              fig1, ax1 = plt.subplots() 
474              ax1.pie(sizes, explode=explode, labels=labels, autopct='%1.1f%%', 
475                      shadow=True, startangle=90) 
476              ax1.axis('equal')  # Equal aspect ratio ensures that pie is drawn as a circle. 
477              plt.savefig(chartfilename) 
478              plt.show()
